	createGrid(rows, collumns):
		grid = new Array(rows)
		for (let i = 0; i < collumns; ++i)
			grid[i] = new Array(collumns)

		for (let i = 0; i < rows; ++i)
			for (let j = 0; j < collumns; ++j)
				grid[i][j] = null;

		return grid
